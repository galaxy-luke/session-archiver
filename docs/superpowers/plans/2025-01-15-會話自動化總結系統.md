# 會話自動化總結系統實施計劃

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 建立一個全局 npm 包 `@claude-code/session-archiver`，使用 AI (glm-4.7-flash) 自動生成 Claude Code 會話總結並歸檔到 Obsidian，支持多專案共用。

**Architecture:** 
- 全局 npm 包架構：單一安裝服務所有專案
- 三層系統：CLI 層 → 核心邏輯層 → AI/資料層
- 常駐守護進程：實時監聽 Obsidian 草稿資料匣，自動歸檔就緒的草稿
- 輕量配置：每個專案只需 `.project-config/session-archiver.json`

**Tech Stack:** 
- Node.js 18+, TypeScript
- CLI: commander.js, inquirer
- AI: @anthropic-ai/sdk (glm-4.7-flash)
- 檔案監聽: chokidar
- Git: simple-git
- 模板: 內建 EJS/Mustache

// src/utils/configManager.ts
import fs from 'fs/promises';
import path from 'path';
import os from 'os';

const PROJECT_CONFIG_DIR = '.project-config';
const CONFIG_FILENAME = 'session-archiver.json';
const GLOBAL_CONFIG_DIR = path.join(os.homedir(), '.session-archiver');
const GLOBAL_CONFIG_FILE = path.join(GLOBAL_CONFIG_DIR, 'config.json');

export class ConfigManager {
  private projectRoot: string;

  constructor(projectRoot: string = process.cwd()) {
    this.projectRoot = projectRoot;
  }

  async ensureProjectConfigDir(): Promise<string> {
    const configDir = path.join(this.projectRoot, PROJECT_CONFIG_DIR);
    await fs.mkdir(configDir, { recursive: true });
    return configDir;
  }

  getProjectConfigPath(): string {
    return path.join(this.projectRoot, PROJECT_CONFIG_DIR, CONFIG_FILENAME);
  }

  async projectConfigExists(): Promise<boolean> {
    try {
      await fs.access(this.getProjectConfigPath());
      return true;
    } catch {
      return false;
    }
  }

  async loadProjectConfig(): Promise<any> {
    const configPath = this.getProjectConfigPath();
    const content = await fs.readFile(configPath, 'utf-8');
    return JSON.parse(content);
  }

  async saveProjectConfig(config: any): Promise<void> {
    const configDir = await this.ensureProjectConfigDir();
    const configPath = path.join(configDir, CONFIG_FILENAME);
    await fs.writeFile(configPath, JSON.stringify(config, null, 2), 'utf-8');
  }

  async loadGlobalConfig(): Promise<GlobalConfig> {
    try {
      const content = await fs.readFile(GLOBAL_CONFIG_FILE, 'utf-8');
      return JSON.parse(content);
    } catch {
      return {}; // Return empty config if doesn't exist
    }
  }

  async mergeConfigs(): Promise<any> {
    const globalConfig = await this.loadGlobalConfig();
    const projectConfig = await this.loadProjectConfig();
    
    // Merge: project config overrides global config
    return {
      ...globalConfig,
      ...projectConfig
    };
  }

  async validateConfig(config: any): Promise<{ valid: boolean; errors: string[] }> {
    const errors: string[] = [];

    if (!config.projectName || typeof config.projectName !== 'string') {
      errors.push('projectName is required and must be a string');
    }

    if (!config.obsidian || !config.obsidian.vaultPath) {
      errors.push('obsidian.vaultPath is required');
    }

    if (!config.ai || !config.ai.model) {
      errors.push('ai.model is required');
    }

    return {
      valid: errors.length === 0,
      errors
    };
  }
}
```

// src/cli/init.ts
import { Command } from 'commander';
import { ConfigManager } from '../utils/configManager';
import { promptForProjectConfig, buildConfigFromAnswers } from '../utils/promptHelper';
import fs from 'fs/promises';

export const initCommand = new Command('init')
  .description('Initialize project configuration')
  .action(async () => {
    try {
      console.log('🚀 Initializing session-archiver for this project...\n');

      const manager = new ConfigManager();

      // Check if config already exists
      if (await manager.projectConfigExists()) {
        const overwrite = await confirmOverwrite();
        if (!overwrite) {
          console.log('❌ Initialization cancelled');
          return;
        }
      }

      // Prompt for configuration
      const answers = await promptForProjectConfig();
      const config = buildConfigFromAnswers(answers);

      // Validate config
      const validation = await manager.validateConfig(config);
      if (!validation.valid) {
        console.error('❌ Configuration validation failed:');
        validation.errors.forEach(err => console.error(`  - ${err}`));
        return;
      }

      // Save config
      await manager.ensureProjectConfigDir();
      await manager.saveProjectConfig(config);

      console.log('\n✅ Configuration saved to:', manager.getProjectConfigPath());
      console.log('\n📝 Next steps:');
      console.log('  1. Review the configuration file');
      console.log('  2. Run: session-archiver daemon start');
      console.log('  3. Start working in Claude Code!');

    } catch (error) {
      console.error('❌ Initialization failed:', error);
      process.exit(1);
    }
  });

async function confirmOverwrite(): Promise<boolean> {
  const inquirer = require('inquirer');
  const { overwrite } = await inquirer.prompt([
    {
      type: 'confirm',
      name: 'overwrite',
      message: 'Configuration already exists. Overwrite?',
      default: false
    }
  ]);
  return overwrite;
}
```

import Anthropic from '@anthropic-ai/sdk';
import { env } from '../utils/env';
import { AIMessage, AISummary, AISummaryRequest } from './types';

export class AnthropicClient {
  private client: Anthropic;

  constructor() {
    this.client = new Anthropic({
      apiKey: env.ANTHROPIC_AUTH_TOKEN()
    });
  }

  async generateSummary(request: AISummaryRequest, model: string = 'glm-4.7-flash'): Promise<AISummary> {
    const prompt = this.buildPrompt(request);

    try {
      const message = await this.client.messages.create({
        model: model,
        max_tokens: 2000,
        temperature: 0.3,
        messages: [
          {
            role: 'user',
            content: prompt
          }
        ]
      });

      return this.parseResponse(message.content[0]);
    } catch (error) {
      console.error('AI generation failed:', error);
      throw error;
    }
  }

  private buildPrompt(request: AISummaryRequest): string {
    return `你是開發會話總結助手。請分析以下會話內容並生成結構化總結：

## 會話資料
- 時間範圍: ${request.startTime} - ${request.endTime}

${request.conversationHistory}

## 檔案變更
${this.formatFileChanges(request.fileChanges)}

## Git 提交記錄
${request.gitCommits}

## 請生成以下內容（Markdown 格式）：

### 1. 會話摘要（2-3 句話）
簡明扼要地總結本次會話的主要目的和成果。

- 每項包含：動詞開頭的描述 + 具體輸出
- 標注優先級（🔴高 / 🟡中 / 🟢低）

### 3. 技術決策與理由

- 解釋為什麼選擇這個方案
- 列出考慮過的替代方案

### 4. 關鍵程式碼片段
- 提取 2-3 個最重要的程式碼變更
- 包含檔案路徑和簡短說明
- 使用語法高亮標記

### 5. 問題與解決方案
- 列出遇到的技術問題
- 說明解決方案和根本原因
- 提取可重用的經驗和教訓

- 3-5 個具體的下一步行動
- 每個行動應該是可執行的
- 可選：標注預估時長和優先級

### 7. 元數據
- 技術棧（使用的技術、框架、工具）
- 相關檔案路徑清單
- 影響範圍評估（低/中/高）
- 風險評估（如適用）

請確保輸出格式清晰、結構化，方便直接填充到模板中。`;
  }

  private formatFileChanges(changes: any[]): string {
    if (!changes || changes.length === 0) return '無檔案變更';
    
    return changes.map(change => 
      `- \`${change.path}\` (${change.changeType})${change.summary ? ': ' + change.summary : ''}`
    ).join('\n');
  }

  private parseResponse(content: any): AISummary {
    const text = content.text;
    const sections: any = {
      摘要: [],
      主要工作項目: [],
      技術決策與理由: [],
      關鍵程式碼片段: [],
      問題與解決方案: [],
      下次行動: [],
      元數據: {}
    };

    const lines = text.split('\n');
    let currentSection: string | null = null;
    let currentSubsection: any[] | null = null;

    for (const line of lines) {
      const sectionMatch = line.match(/^###\s+(\d+)\.\s+(.+)/);
      if (sectionMatch) {
        const sectionNumber = parseInt(sectionMatch[1]);
        const sectionNames = ['會話摘要', '主要工作項目', '技術決策與理由', '關鍵程式碼片段', '問題與解決方案', '下次行動', '元數據'];
        currentSection = sectionNames[sectionNumber - 1];
        if (sectionNumber < 7) {
          currentSubsection = sections[currentSection];
        }
        continue;
      }

      if (currentSection && currentSubsection && line.trim()) {
        currentSubsection.push(line.trim());
      }
    }

    return sections;
  }
}

import simpleGit from 'simple-git';
import { FileChange } from '../ai/types';

export class FileCollector {
  async collectFileChanges(workDir: string): Promise<FileChange[]> {
    const git = simpleGit(workDir);
    
    try {
      const status = await git.status();
      const changes: FileChange[] = [];

      // Modified files
      for (const file of status.modified) {
        changes.push({
          path: file,
          changeType: 'modified',
          summary: await this.generateDiffSummary(workDir, file)
        });
      }

      // New files
      for (const file of status.created) {
        changes.push({
          path: file,
          changeType: 'created'
        });
      }

      // Deleted files
      for (const file of status.deleted) {
        changes.push({
          path: file,
          changeType: 'deleted'
        });
      }

      return changes;
    } catch (error) {
      console.error('Failed to collect file changes:', error);
      return [];
    }
  }

  private async generateDiffSummary(workDir: string, file: string): Promise<string> {
    const git = simpleGit(workDir);
    try {
      const diff = await git.diff(['--unified=0', file]);
      // Simple summary: count changed lines
      const addedLines = (diff.match(/^\+/gm) || []).length;
      const removedLines = (diff.match(/^-/gm) || []).length;
      return `+${addedLines} -${removedLines} lines`;
    } catch {
      return 'diff unavailable';
    }
  }
}

  async getGitStatus(workDir: string): Promise<string> {
    const git = simpleGit(workDir);
    try {
      const status = await git.status();
      return status.files.map(f => `${f.index} ${f.working} ${f.path}`).join('\n') || 'clean';
    } catch {
      return 'unavailable';
    }
  }

  async getRecentCommits(workDir: string, count: number = 10): Promise<string> {
    const git = simpleGit(workDir);
    try {
      const log = await git.log({ maxCount: count });
      return log.all.map(commit => `${commit.hash.substring(0, 7)} ${commit.message}`).join('\n');
    } catch {
      return 'unavailable';
    }
  }

  async collect(workDir: string): Promise<any> {
    return {
      gitStatus: await this.getGitStatus(workDir),
      gitCommits: await this.getRecentCommits(workDir)
    };
  }
}

// src/core/generator.ts
import { AnthropicClient } from '../ai/anthropicClient';
import { SessionCollector, FileCollector, GitCollector } from '../collectors';
import { ConfigManager } from '../utils/configManager';
import { SessionContext } from '../collectors/sessionCollector';
import fs from 'fs/promises';
import path from 'path';
import { readFileSync } from 'fs';
import { format } from 'date-fns';

export class Generator {
  private aiClient: AnthropicClient;
  private sessionCollector: SessionCollector;
  private fileCollector: FileCollector;
  private gitCollector: GitCollector;
  private configManager: ConfigManager;

  constructor() {
    this.aiClient = new AnthropicClient();
    this.sessionCollector = new SessionCollector();
    this.fileCollector = new FileCollector();
    this.gitCollector = new GitCollector();
    this.configManager = new ConfigManager();
  }

  async generate(context: SessionContext): Promise<string> {
    // 1. Collect all data
    const sessionData = await this.sessionCollector.collect(context);
    const fileChanges = await this.fileCollector.collectFileChanges(context.workDirectory);
    const gitData = await this.gitCollector.collect(context.workDirectory);

    // 2. Get AI summary
    const summary = await this.aiClient.generateSummary({
      conversationHistory: sessionData.conversationHistory,
      fileChanges: fileChanges,
      gitStatus: gitData.gitStatus,
      gitCommits: gitData.gitCommits,
      workDirectory: context.workDirectory,
      startTime: context.startTime.toISOString(),
      endTime: new Date().toISOString()
    });

    // 3. Load config
    const config = await this.configManager.mergeConfigs();
    const model = config.ai?.model || 'glm-4.7-flash';

    // 4. Render template
    const draftPath = await this.renderDraft(summary, sessionData, model);

    return draftPath;
  }

  private async renderDraft(summary: any, sessionData: any, model: string): Promise<string> {
    const templatePath = path.join(__dirname, '../../templates/session-draft.md');
    let template = readFileSync(templatePath, 'utf-8');

    // Calculate duration
    const startTime = new Date(sessionData.startTime);
    const endTime = new Date(sessionData.endTime);
    const duration = this.calculateDuration(startTime, endTime);

    // Replace placeholders
    template = template.replace(/\{\{ date \}\}/g, format(new Date(), 'yyyy-MM-dd'));
    template = template.replace(/\{\{ startTime \}\}/g, format(startTime, 'HH:mm'));
    template = template.replace(/\{\{ endTime \}\}/g, format(endTime, 'HH:mm'));
    template = template.replace(/\{\{ duration \}\}/g, duration);
    template = template.replace(/\{\{ model \}\}/g, model);
    template = template.replace(/\{\{ complexity \}\}/g, '0.5');
    template = template.replace(/\{\{ summary \}\}/g, summary.摘要.join('') || '暂无摘要');
    template = template.replace(/\{\{ workItems \}\}/g, summary.主要工作項目.join('\n') || '- 無主要工作項目');
    template = template.replace(/\{\{ technicalDecisions \}\}/g, summary.技術決策與理由.join('\n') || '無重大技術決策');
    template = template.replace(/\{\{ codeSnippets \}\}/g, summary.關鍵程式碼片段.join('\n') || '無關鍵程式碼變更');
    template = template.replace(/\{\{ problemsAndSolutions \}\}/g, summary.問題與解決方案.join('\n') || '無遇到問題');
    template = template.replace(/\{\{ nextActions \}\}/g, summary.下次行動.join('\n') || '- 無特定行動項');
    template = template.replace(/\{\{ relatedFiles \}\}/g, '');
    template = template.replace(/\{\{ relatedResources \}\}/g, '');
    template = template.replace(/\{\{ techStack \}\}/g, summary.元數據.技術棧 || 'N/A');
    template = template.replace(/\{\{ relatedFilesList \}\}/g, summary.元數據.相關檔案路徑?.join(', ') || 'N/A');
    template = template.replace(/\{\{ impactScope \}\}/g, summary.元數據.影響範圍評估 || '中');
    template = template.replace(/\{\{ riskAssessment \}\}/g, summary.元數據.風險評估 || '無');

    // Generate filename
    const timestamp = format(new Date(), 'yyyy-MM-dd-HHmm');
    const filename = `draft-${timestamp}.md`;

    // Save to Obsidian drafts
    const config = await this.configManager.mergeConfigs();
    const vaultPath = config.obsidian?.vaultPath;
    if (!vaultPath) {
      throw new Error('Obsidian vault path not configured');
    }

    const draftsPath = path.join(vaultPath, config.obsidian?.draftsPath || '草稿');
    await fs.mkdir(draftsPath, { recursive: true });

    const draftPath = path.join(draftsPath, filename);
    await fs.writeFile(draftPath, template, 'utf-8');

    return draftPath;
  }

  private calculateDuration(start: Date, end: Date): string {
    const diffMs = end.getTime() - start.getTime();
    const diffMins = Math.floor(diffMs / 60000);
    const hours = Math.floor(diffMins / 60);
    const mins = diffMins % 60;
    
    if (hours > 0) {
      return `${hours}h ${mins}m`;
    }
    return `${mins}m`;
  }
}

    console.log(`📡 正在監聽: ${draftsPath}/*.md`);

    // Setup file watcher
    this.watcher = chokidar.watch(path.join(draftsPath, '*.md'), {
      persistent: true,
      ignoreInitial: true,
      awaitWriteFinish: {
        stabilityThreshold: 2000,
        pollInterval: 100
      }
    });

    // Event handlers
    this.watcher
      .on('change', (filePath) => this.handleFileChange(filePath))
      .on('add', (filePath) => this.handleFileChange(filePath))
      .on('ready', () => console.log('✅ 監聽器就緒'))
      .on('error', (error) => this.handleError(error));

    // Wait for ready
    await new Promise(resolve => this.watcher?.once('ready', resolve));

    // Initial scan
    await this.initialScan(draftsPath);

    // Periodic check
    const checkInterval = config.daemon?.checkInterval || 30000;
    this.startPeriodicCheck(checkInterval);

    // Graceful shutdown
    this.setupGracefulShutdown();

    // Update status
    await this.updateStatus();
  }

  private async handleFileChange(filePath: string): Promise<void> {
    if (this.isProcessing) {
      console.log('⏳ 正在處理中，跳過本次觸發');
      return;
    }

    try {
      this.isProcessing = true;

      const content = await fs.readFile(filePath, 'utf-8');
      const status = this.extractStatus(content);

      if (status === 'ready') {
        console.log(`📋 檢測到就緒草稿: ${path.basename(filePath)}`);
        await this.archiveDraft(filePath);
        this.stats.processedCount++;
      }
    } catch (error) {
      console.error('❌ 處理檔案變更時出錯:', error);
      this.stats.errorCount++;
    } finally {
      this.isProcessing = false;
      await this.updateStatus();
    }
  }

  private extractStatus(content: string): string | null {
    const match = content.match(/^status:\s*(.+)$/m);
    return match ? match[1].trim() : null;
  }

  private async initialScan(draftsPath: string): Promise<void> {
    console.log('🔍 執行初始掃描...');
    try {
      const files = await fs.readdir(draftsPath);
      for (const file of files.filter(f => f.endsWith('.md'))) {
        await this.handleFileChange(path.join(draftsPath, file));
      }
      console.log('✅ 初始掃描完成');
    } catch (error) {
      console.error('❌ 初始掃描失敗:', error);
    }
  }

  private startPeriodicCheck(interval: number): void {
    this.checkInterval = setInterval(async () => {
      this.stats.lastCheck = new Date().toISOString();
      await this.periodicCheck();
      await this.updateStatus();
    }, interval);
  }

  private async periodicCheck(): Promise<void> {
    const config = await this.configManager.mergeConfigs();
    const vaultPath = config.obsidian?.vaultPath;
    const draftsPath = path.join(vaultPath!, config.obsidian?.draftsPath || '草稿');

    try {
      const files = await fs.readdir(draftsPath);
      for (const file of files.filter(f => f.endsWith('.md'))) {
        const filePath = path.join(draftsPath, file);
        const content = await fs.readFile(filePath, 'utf-8');
        const status = this.extractStatus(content);

        if (status === 'ready') {
          await this.archiveDraft(filePath);
          this.stats.processedCount++;
        }
      }
    } catch (error) {
      console.error('❌ 定期檢查時出錯:', error);
      this.stats.errorCount++;
    }
  }

  private async archiveDraft(draftPath: string): Promise<void> {
    console.log(`📦 開始歸檔: ${path.basename(draftPath)}`);
    
    try {
      const result = await this.archiver.archive(draftPath);
      console.log(`✅ 已歸檔到: ${result.targetPath}`);

      // Send notification if enabled
      const config = await this.configManager.mergeConfigs();
      if (config.daemon?.notifications) {
        this.sendNotification(`歸檔完成: ${path.basename(result.targetPath)}`);
      }
    } catch (error) {
      console.error('❌ 歸檔失敗:', error);
      await this.markAsError(draftPath, (error as Error).message);
    }
  }

  private async markAsError(draftPath: string, errorMessage: string): Promise<void> {
    try {
      let content = await fs.readFile(draftPath, 'utf-8');
      content = content.replace(
        /^status:\s*ready$/m,
        `status: error\nerror: ${errorMessage}`
      );
      await fs.writeFile(draftPath, content, 'utf-8');
    } catch (error) {
      console.error('❌ 標記錯誤狀態失敗:', error);
    }
  }

  private sendNotification(message: string): void {
    notifier.notify({
      title: '會話歸檔系統',
      message: message
    });
  }

  private handleError(error: Error): void {
    console.error('❌ 監聽器錯誤:', error);
    this.stats.errorCount++;
  }

  private async updateStatus(): Promise<void> {
    try {
      await fs.writeFile(this.statusPath, JSON.stringify(this.stats, null, 2), 'utf-8');
    } catch (error) {
      console.error('❌ 更新狀態失敗:', error);
    }
  }

  private setupGracefulShutdown(): void {
    const shutdown = async () => {
      console.log('🛑 正在關閉守護進程...');

      if (this.watcher) {
        await this.watcher.close();
      }

      if (this.checkInterval) {
        clearInterval(this.checkInterval);
      }

      // Clean up status file
      try {
        await fs.unlink(this.statusPath);
      } catch {
        // Ignore if already deleted
      }

      console.log('✅ 守護進程已關閉');
      process.exit(0);
    };

    process.on('SIGINT', shutdown);
    process.on('SIGTERM', shutdown);
  }

  async stop(): Promise<void> {
    // Trigger graceful shutdown
    process.kill(process.pid, 'SIGTERM');
  }

  getStatus(): DaemonStatus {
    return this.stats;
  }
}

import fs from 'fs/promises';

const PID_FILE = '.daemon-pid';

export class DaemonManager {
  async ensureRunning(): Promise<boolean> {
    // Check if daemon already running
    const existingPid = await this.getPid();
    if (existingPid && await this.isProcessRunning(existingPid)) {
      console.log('✅ 守護進程已在運行 (PID:', existingPid, ')');

  async start(): Promise<boolean> {
    console.log('🚀 啟動新的守護進程...');
    
    const daemon = new DaemonCore();
    await daemon.start();
    
    // Save PID
    await this.savePid(process.pid);
    
    return true;
  }

  async stop(): Promise<void> {
    const pid = await this.getPid();
    if (!pid) {
      console.log('❌ 守護進程未運行');
      return;
    }

    if (await this.isProcessRunning(pid)) {
      process.kill(pid, 'SIGTERM');
      console.log('✅ 守護進程已停止');
    }

    await this.clearPid();
  }

  async restart(): Promise<void> {
    await this.stop();
    // Wait a bit for clean shutdown
    await new Promise(resolve => setTimeout(resolve, 1000));
    await this.start();
  }

  async status(): Promise<void> {
    const pid = await this.getPid();
    if (!pid) {
      console.log('❌ 守護進程未運行');
      return;
    }

    if (await this.isProcessRunning(pid)) {
      console.log('✅ 守護進程運行中 (PID:', pid, ')');
      
      // Try to read status file
      try {
        const statusPath = path.join(process.cwd(), '.daemon-status.json');
        const statusContent = await fs.readFile(statusPath, 'utf-8');
        const status = JSON.parse(statusContent);
        
        console.log('\n📊 統計資訊:');
        console.log(`  - 運行時間: ${Math.floor(process.uptime() / 60)} 分鐘`);
        console.log(`  - 已處理: ${status.processedCount} 次`);
        console.log(`  - 錯誤數: ${status.errorCount}`);
        console.log(`  - 最後檢查: ${status.lastCheck || 'N/A'}`);
        console.log(`  - 記憶體使用: ${Math.round(process.memoryUsage().heapUsed / 1024 / 1024)}MB`);
      } catch {
        console.log('⚠️  無法讀取詳細狀態');
      }
    } else {
      console.log('❌ 守護進程未運行（PID 文件存在但進程不存在）');
      await this.clearPid();
    }
  }

  private async getPid(): Promise<number | null> {
    try {
      const pidContent = await fs.readFile(PID_FILE, 'utf-8');
      return parseInt(pidContent.trim());
    } catch {
      return null;
    }
  }

  private async savePid(pid: number): Promise<void> {
    await fs.writeFile(PID_FILE, pid.toString(), 'utf-8');
  }

  private async clearPid(): Promise<void> {
    try {
      await fs.unlink(PID_FILE);
    } catch {
      // Ignore
    }
  }

  private async isProcessRunning(pid: number): Promise<boolean> {
    try {
      process.kill(pid, 0); // Signal 0 just checks if process exists
      return true;
    } catch {
      return false;
    }
  }
}

import { format } from 'date-fns';

export interface ArchiveResult {
  targetPath: string;
  originalPath: string;
  newFilename: string;
}

export class FileNamer {
  generateFileName(content: string, projectName?: string): string {
    // 1. Extract date
    const dateMatch = content.match(/date:\s*(\d{4}-\d{2}-\d{2})/);
    const date = dateMatch ? dateMatch[1] : format(new Date(), 'yyyy-MM-dd');

    // 2. Extract title from summary
    const summaryMatch = content.match(/>\s*\[!summary\][^>]*>([^*\n]+)/);
    let title = summaryMatch ? summaryMatch[1].trim() : '';

    // 3. Extract from work items if no summary
    if (!title) {
      const workItemMatch = content.match(/-\s*🔴\s*\*\*[^*]+\*\*:\s*([^\n]+)/);
      if (workItemMatch) {
        title = workItemMatch[1].trim();
      }
    }

    // 4. Use project name if still no title
    if (!title && projectName) {
      title = `${projectName}會話記錄`;
    }

    // 5. Fallback
    if (!title) {
      title = '會話記錄';
    }

    // 6. Clean and truncate
    title = this.cleanTitle(title);

    return `${date}-${title}.md`;
  }

  private cleanTitle(title: string): string {
    // Remove special characters
    title = title.replace(/[<>:"/\\|?*]/g, '');
    
    // Remove markdown formatting
    title = title.replace(/\*\*/g, '');
    title = title.replace(/\[|\]/g, '');
    
    // Truncate to max 50 characters
    if (title.length > 50) {
      title = title.substring(0, 47) + '...';
    }

    return title;
  }

  determineTargetPath(projectTag: string | null, config: any): string {
    if (!projectTag) {
      // General diary
      return path.join(config.obsidian.vaultPath, config.obsidian.dailyNotesPath);
    }

    // Extract project name from tag (e.g., #AIT-Digital -> AIT-Digital)
    const projectName = projectTag.replace(/^#/, '');
    return path.join(config.obsidian.vaultPath, config.obsidian.projectPath.replace('${projectName}', projectName), '會話日誌');
  }

// src/core/archiver.ts
import fs from 'fs/promises';
import path from 'path';
import { ConfigManager } from '../utils/configManager';
import { FileNamer, ArchiveResult } from './namer';

export class Archiver {
  private configManager: ConfigManager;
  private namer: FileNamer;

  constructor() {
    this.configManager = new ConfigManager();
    this.namer = new FileNamer();
  }

  async archive(draftPath: string): Promise<ArchiveResult> {
    // 1. Read draft content
    const content = await fs.readFile(draftPath, 'utf-8');

    // 2. Parse frontmatter
    const { status, project, ...metadata } = this.parseFrontmatter(content);

    if (status !== 'ready') {
      throw new Error('Draft is not ready for archiving');
    }

    // 3. Get config
    const config = await this.configManager.mergeConfigs();

    // 4. Generate filename
    const projectName = config.projectName;
    const filename = this.namer.generateFileName(content, projectName);

    // 5. Determine target path
    const targetDir = this.namer.determineTargetPath(project, config);
    const targetPath = path.join(targetDir, filename);

    // 6. Create target directory
    await fs.mkdir(targetDir, { recursive: true });

    // 7. Move file
    await fs.rename(draftPath, targetPath);

    // 8. Update frontmatter
    await this.updateArchivedFrontmatter(targetPath, metadata);

    return {
      targetPath,
      originalPath: draftPath,
      newFilename: filename
    };
  }

  private parseFrontmatter(content: string): any {
    const frontmatter: any = {};
    const lines = content.split('\n');
    let inFrontmatter = false;

    for (const line of lines) {
      if (line.trim() === '---') {
        inFrontmatter = !inFrontmatter;
        continue;
      }

      if (inFrontmatter) {
        const match = line.match(/^(\w+):\s*(.+)$/);
        if (match) {
          frontmatter[match[1]] = match[2].trim();
        }
      }
    }

    return frontmatter;
  }

  private async updateArchivedFrontmatter(filePath: string, metadata: any): Promise<void> {
    let content = await fs.readFile(filePath, 'utf-8');

    // Update status
    content = content.replace(/^status:\s*draft$/m, 'status: archived');

    // Add archived timestamp
    const archivedAt = `archivedAt: ${new Date().toISOString()}`;
    content = content.replace(/^---$/m, `${archivedAt}\n---`);

    await fs.writeFile(filePath, content, 'utf-8');
  }
}

// src/utils/claudeSetup.ts
import fs from 'fs/promises';
import path from 'path';
import os from 'os';

export async function setupClaudeHooks(): Promise<void> {
  const claudeDir = path.join(os.homedir(), '.claude');
  const settingsPath = path.join(claudeDir, 'settings.json');

  try {
    // Read existing settings
    let settings: any = {};
    try {
      const content = await fs.readFile(settingsPath, 'utf-8');
      settings = JSON.parse(content);
    } catch {
      // File doesn't exist, start fresh
    }

    // Add hooks
    if (!settings.hooks) {
      settings.hooks = {};
    }

    settings.hooks.sessionEnd = {
      command: 'session-archiver',
      args: ['generate'],
      enabled: true
    };

    settings.hooks.sessionStart = {
      command: 'session-archiver',
      args: ['daemon', 'ensure'],
      enabled: true
    };

    // Add env if not exists
    if (!settings.env) {
      settings.env = {};
    }

    if (!settings.env.ANTHROPIC_AUTH_TOKEN) {
      settings.env.ANTHROPIC_AUTH_TOKEN = 'your_token_here';
      settings.env.ANTHROPIC_DEFAULT_HAIKU_MODEL = 'glm-4.7-flash';
    }

    // Write settings
    await fs.mkdir(claudeDir, { recursive: true });
    await fs.writeFile(settingsPath, JSON.stringify(settings, null, 2), 'utf-8');

    console.log('✅ Claude Code hooks configured');
    console.log(`   Settings file: ${settingsPath}`);
    console.log('\n⚠️  Please update ANTHROPIC_AUTH_TOKEN in the settings file');
  } catch (error) {
    throw new Error(`Failed to setup Claude hooks: ${error}`);
  }
}
